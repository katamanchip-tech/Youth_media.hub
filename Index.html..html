function switchTab(target) {  
['media', 'jobs', 'reels', 'yt'].forEach(s => {  
document.getElementById('sec-' + s).classList.toggle('hidden', s !== target);  
document.getElementById('btn-' + s).classList.toggle('tab-active', s === target);  
});  
if(target === 'reels') renderReels();  
}
